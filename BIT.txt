struct BIT{
	int N;
	vector<long long> A;
	
	void init(int n){
		N = n;
		A = vector<long long>(N+1, 0);
	}
	
	void update(int loc, long long x){
		if(loc < 0)return;
		while(loc < N){
			A[loc] += x;
			loc |= (loc + 1);
		}
	}
	
	long long query(int loc){
		long long sum=0;
		while(loc >= 0){
			sum += A[loc];
			loc = (loc & (loc+1)) - 1;
		}
		return sum;
	}
	
	long long query(int L, int R){
		return query(L) - query(R-1);
	}
	// 0 based - indexing
};
